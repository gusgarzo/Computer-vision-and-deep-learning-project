\section{Background \& Motivation}
% =======================================================

\begin{frame}{Background \& Motivation}
  \begin{columns}[T,totalwidth=\textwidth]
    \begin{column}{0.58\textwidth}
      \small
      \begin{itemize}
        \item Autonomous driving systems perform well in ideal road conditions, but snowy roads remain challenging because boundaries and lane markings become hard to see.
        \item In winter, \textbf{snow poles} provide a robust visual cue for road localisation because they remain visible even when the road surface is covered.
        \item Our project therefore focuses on \textbf{single-class object detection} of snow poles in realistic winter driving scenes from Trondheim.
      \end{itemize}
      \vspace{0.5em}
      \textbf{Main challenge:} poles are thin, visually small, and often far from the camera.
      \vspace{0.5em}
      
      The final goal is not only accuracy, but also a detector that could realistically support \textbf{real-time edge deployment}.
    \end{column}
    \begin{column}{0.38\textwidth}
      \begin{block}{Problem framing}
        \small
        \begin{itemize}
          \item task: one-class object detection
          \item application: winter road localisation
          \item constraint: edge-ready, real-time inference
          \item evaluation: Precision, Recall, mAP@50 and mAP@0.5:0.95
        \end{itemize}
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\section{Approach / Strategy}
% =======================================================

\begin{frame}{Approach / Strategy}
  \begin{columns}[T,totalwidth=\textwidth]
    \begin{column}{0.58\textwidth}
      \begin{block}{Project strategy}
        \small
        \begin{itemize}
          \item We first treated the task as a \textbf{dataset understanding problem}, not only a model training problem.
          \item We analysed \textbf{both available leaderboards separately}: \texttt{Road\_poles\_iPhone} and \texttt{roadpoles\_v1}.
          \item The goal of the EDA was to understand whether both datasets had similar object scale, geometry and scene structure.
        \end{itemize}
      \end{block}
      \begin{alertblock}{Why this matters}
        \small
        The course brief explicitly asks for model development to be guided by exploratory data analysis, so this step is part of the methodology rather than a side note.
      \end{alertblock}
    \end{column}
    \begin{column}{0.38\textwidth}
      \begin{block}{Pipeline}
        \small
        \centering
        \vspace{0.3em}
        \fbox{\parbox{0.84\linewidth}{\centering
        Problem framing\\[0.2em]
        $\downarrow$\\[0.2em]
        EDA on both datasets\\[0.2em]
        $\downarrow$\\[0.2em]
        Resolution and model selection\\[0.2em]
        $\downarrow$\\[0.2em]
        Training, validation and leaderboard submission}}
        \vspace{0.3em}
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Approach / Strategy -- Key Reasoning}
  \begin{block}{What the EDA needed to answer}
    \begin{itemize}
      \item Are the two datasets similar enough that we should interpret them as the same task?
      \item Is the main difficulty object density, object scale, object geometry, or scene structure?
      \item Which data properties are most likely to explain differences in final leaderboard performance?
    \end{itemize}
  \end{block}

  \begin{block}{How the analysis informs the rest of the presentation}
    \begin{itemize}
      \item If poles are very small, input resolution becomes a critical design choice.
      \item If one dataset is geometrically harder, lower results there do not necessarily mean the model is worse overall.
      \item This gives context for the model and training decisions shown in the next section.
    \end{itemize}
  \end{block}
\end{frame}

\section{Methods \& Models}
% =======================================================

\begin{frame}{Overview -- Two Architectures}
  \begin{columns}
    \begin{column}{0.48\textwidth}
      \begin{block}{Model 1 -- YOLOv8n Baseline}
        \begin{itemize}
          \item Single-stage detector
          \item You Only Look Once -- one forward pass
          \item Lightweight baseline for fast iteration
          \item Separate notebooks prepared for both datasets
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{block}{Model 2 -- Faster R-CNN}
        \begin{itemize}
          \item Two-stage detector
          \item Region Proposal Network + classification
          \item Torchvision pretrained backbones + FPN
          \item Strongest validated results on both datasets
        \end{itemize}
      \end{block}
    \end{column}
  \end{columns}
  \vspace{0.5em}
  \centering
  \textit{Both model families were used on both leaderboards: \texttt{Road\_poles\_iPhone} and \texttt{roadpoles\_v1}}
\end{frame}

% -------------------------------------------------------
\begin{frame}[shrink=2]{Model 1 -- YOLO Baseline Architecture}
  \begin{columns}
    \begin{column}{0.55\textwidth}
      \textbf{Why YOLO as a baseline?}
      \begin{itemize}
        \item Single-stage detector with very fast training and inference
        \item Good baseline for checking whether the task can be solved with a lightweight model
        \item Easy to train and export with the Ultralytics workflow
        \item Relevant to the deployment goal because it is edge-friendly
      \end{itemize}
      \vspace{0.5em}
      \textbf{Notebook setup used in the repo:}
      \begin{itemize}
        \item \texttt{yolov8n.pt} pretrained checkpoint
        \item 100 epochs
        \item \texttt{imgsz=640}
        \item Separate notebooks prepared for each dataset
      \end{itemize}
    \end{column}
    \begin{column}{0.42\textwidth}
      \begin{block}{Model stats}
        \begin{itemize}
          \item Variant: YOLOv8n
          \item Input size: $640\times640$
          \item Classes: 1 (pole)
          \item Pretrained: COCO
          \item Device: CUDA in notebook config
          \item Role in project: baseline detector
        \end{itemize}
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

% -------------------------------------------------------
\begin{frame}{Model 2 -- Faster R-CNN Architecture}
  \begin{columns}
    \begin{column}{0.55\textwidth}
      \textbf{Two-stage detection pipeline:}
      \begin{enumerate}
        \item \textbf{Stage 1 -- RPN:} Scans feature map, proposes candidate regions containing objects
        \item \textbf{Stage 2 -- ROI Head:} Classifies each proposal and refines the bounding box
      \end{enumerate}
      \vspace{0.5em}
      \textbf{FPN backbone:} Extracts features at multiple scales simultaneously -- important for poles at different distances
      \vspace{0.5em}
      \textbf{Key difference from YOLO:} Two passes = slower, but better suited to careful localisation of thin objects
    \end{column}
    \begin{column}{0.42\textwidth}
      \begin{block}{Architecture}
        \begin{itemize}
          \item Backbone family: pretrained Faster R-CNN + FPN
          \item Pretrained: COCO
          \item Head: replaced for 2 classes (background + pole)
          \item Optimiser: AdamW
          \item Scheduler: CosineAnnealingLR
          \item Epochs: 100
        \end{itemize}
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

% -------------------------------------------------------
\begin{frame}[fragile,shrink=5]{Model 2 -- Faster R-CNN Training Config}
  \begin{block}{Custom dataset class -- YOLO labels to Faster R-CNN format}
\begin{lstlisting}[language=Python]
# Convert normalised YOLO format to absolute pixel coords
x1 = (cx - bw/2) * W
y1 = (cy - bh/2) * H
x2 = (cx + bw/2) * W
y2 = (cy + bh/2) * H
# Labels: 0 = background, 1 = pole
\end{lstlisting}
  \end{block}
  \begin{block}{Augmentations (train only)}
\begin{lstlisting}[language=Python]
train_transform = T.Compose([
    T.ToTensor(),
    T.ColorJitter(brightness=0.4, contrast=0.4,
                  saturation=0.3, hue=0.1),
    T.RandomHorizontalFlip(p=0.5),
])
\end{lstlisting}
  \end{block}
\end{frame}

\section{Discussion}
% =======================================================

\begin{frame}{Discussion -- Why Faster R-CNN Performed Best}
  \begin{block}{Three key factors}
    \begin{enumerate}
      \item \textbf{Two-stage localisation:} proposing regions first and refining them afterwards seems to help on thin poles.
      \item \textbf{Multi-scale features:} the FPN-based Faster R-CNN pipeline is well aligned with small and distant roadside objects.
      \item \textbf{EDA consistency:} the same model performed better on the easier dataset and worse on the harder one, which makes the results more interpretable.
    \end{enumerate}
  \end{block}
  \vspace{0.3em}
  \textit{In our completed runs, Faster R-CNN was the strongest validated approach on both datasets, even though it is slower than YOLO.}
\end{frame}

% -------------------------------------------------------
\begin{frame}{Discussion -- Limitations \& Future Work}
  \begin{columns}
    \begin{column}{0.48\textwidth}
      \textbf{Current limitations:}
      \begin{itemize}
        \item \texttt{roadpoles\_v1} remains much harder than \texttt{Road\_poles\_iPhone}
        \item Hidden test gap still suggests domain shift (weather, lighting)
        \item Faster R-CNN is accurate, but slower for real-time edge deployment
        \item No pseudo-labelling of unlabelled RoadPoles-MSJ data
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{Future improvements:}
      \begin{itemize}
        \item Run a cleaner hyperparameter sweep for both architectures
        \item Pseudo-label RoadPoles-MSJ with best YOLO model
        \item Try lighter Faster R-CNN backbones for faster inference
        \item Test time augmentation (TTA) for better test scores
        \item Ensemble YOLO + Faster R-CNN predictions
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}

\section{Key Learning Points}
% =======================================================

\begin{frame}[shrink=4]{Key Learning Points}
  \begin{block}{1 -- EDA predicted the ranking between datasets}
    The EDA suggested that \texttt{roadpoles\_v1} would be the harder benchmark because the poles are smaller and more elongated. The Faster R-CNN results later confirmed exactly that pattern.
  \end{block}
  \begin{block}{2 -- Understanding two-stage detection}
    Implementing Faster R-CNN manually revealed how the RPN and ROI pooling work together. The label format conversion from YOLO normalised boxes to absolute pixel coordinates highlighted how different frameworks expect data.
  \end{block}
  \begin{block}{3 -- Applied deep learning is mostly engineering}
    A significant portion of the project was path issues, yaml configuration, CUDA driver mismatches, and dataset format debugging. These are the real-world challenges in any CV deployment scenario.
  \end{block}
\end{frame}
